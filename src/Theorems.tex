\newcommand\drawLeftLine[2][-1ex]{ % #1 = vertical shift, #2 = the color
    \draw[color = #2, line~width = \JHTheorems_line_width, opacity = 0.7]
        ([ shift = { (\JHTheorems_line_half_width, #1) } ] interior.north~west )
    -- ([ shift = { (\JHTheorems_line_half_width, 0.8em) } ] interior.south~west );
}
\newcommand\drawLeftBottomLine[2][-1ex]{ % #1 = vertical shift, #2 = the color
    \draw[color = #2, line~width = \JHTheorems_line_width, opacity = 0.7]
        ([ shift = { (\JHTheorems_line_half_width, #1) } ] interior.north~west )
    -- ([ shift = { (\JHTheorems_line_half_width, 0pt) } ] interior.south~west )
    -- ([ shift = { (-1em, 0pt) } ] interior.south~east );
}

\newtcolorbox{leftLineBox}[2][]{
    enhanced,
    use~color~stack, % This fixes the coloring problem (i.e. that the page breaks reverted color changes), see https://github.com/T-F-S/tcolorbox/issues/110
    empty,
    breakable=true,
    left=7pt,
    parbox=false, % this fixes the indentation of paragraphs
    overlay~unbroken~and~first={ \drawLeftLine[-4ex]{#2} },
    overlay~middle~and~last={ \drawLeftLine{#2} },
    #1
}

\newtcolorbox{leftBottomLineBox}[2][]{
    enhanced,
    use~color~stack, % This fixes the coloring problem (i.e. that the page breaks reverted color changes), see https://github.com/T-F-S/tcolorbox/issues/110
    empty,
    breakable=true,
    left=7pt,
    parbox=false, % this fixes the indentation of paragraphs
    overlay~first={ \drawLeftLine[-4ex]{#2} },
    overlay~middle={ \drawLeftLine{#2} },
    overlay~last={ \drawLeftBottomLine{#2} },
    overlay~unbroken={ \drawLeftBottomLine[-4ex]{#2} },
    #1
}

\NewDocumentCommand{\DeclareLinedTheorem}{o m m}{ 
    % #1 = options for the theorem, #2 = name of the theorem, #3 = color
    \declaretheorem[#1]{internal_#2}
    \NewDocumentEnvironment{#2}{o d()}{
        % #1 = optional argument for the theorem, #2 = optional argument for the color
        \IfValueTF{##2}{
            \def\JHTheorems_current_color{##2}
            \begin{leftLineBox}{##2}
        }{
            \def\JHTheorems_current_color{#3}
            \begin{leftLineBox}{#3}
        }
        \IfValueTF{##1}{
            \begin{internal_#2}[##1]
        }{
            \begin{internal_#2}
        }
    }{
        \end{internal_#2}
        \end{leftLineBox}
    }
}

\NewDocumentCommand{\DeclareBoxLinedTheorem}{o m m}{ 
    \declaretheorem[#1]{internal_#2}
    \NewDocumentEnvironment{#2}{o d()}{
        \IfValueTF{##2}{
            \def\JHTheorems_current_color{##2}
            \begin{leftBottomLineBox}{##2}
        }{
            \def\JHTheorems_current_color{#3}
            \begin{leftBottomLineBox}{#3}
        }
        \IfValueTF{##1}{
            \begin{internal_#2}[##1]
        }{
            \begin{internal_#2}
        }
    }{
        \end{internal_#2}
        \end{leftBottomLineBox}
    }
}

\newcommand{\FancyHead}[1]{
    \sbox{\JHTheoremsHeadBox}{#1}%
    \begin{tikzpicture}[baseline=(head.base)]
        \draw[draw=\JHTheorems_current_color, line~width=\JHTheorems_line_width, opacity=0.7]
            (0,-0.5ex) -- (\wd\JHTheoremsHeadBox,-0.5ex);
        \node[inner~sep=0pt,outer~sep=0pt,anchor=base~west] (head) at (0,0) {\usebox{\JHTheoremsHeadBox}};
    \end{tikzpicture}%
}
