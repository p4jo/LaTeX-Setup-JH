                \tl_gput_left:Nx\autoindex_current_tl
            }{
                \tl_gput_left:Nx\autoindex_current_tl!

            > { \SplitList{ ? } } m

                \tl_clear:N\autoindex_current_tl
                \ProcessList{#1}{\autoindex_add_superindex:n}
